🛠️ Implementation Guide

Installation & Setup

<!-- Include Grove Components -->
<script src="{% static 'core/js/components/grove/grove-components.js' %}"></script>

<!-- Auto-initialize -->
<script>
  document.addEventListener('DOMContentLoaded', () => {
    GroveComponents.instance.init({
      theme: 'light',
      animation: true
    });
  });
</script>

Usage Examples

<!-- Auto-initialization with data attributes -->
<div data-grove="GroveButton" 
     data-grove-config='{"variant":"primary","size":"lg"}'>
  Click me
</div>

<!-- Manual initialization -->
<script>
  const button = Grove.createComponent('GroveButton', element, {
    variant: 'primary',
    onClick: () => console.log('Clicked!')
  });
</script>
